direx.exe <path> <action> [result]

  action:

    all - 
		для всех файлов запоминает путь к файлу, 
		относительно параметра "стартовая папка"
    срр - 
		для всех файлов с расширением срр (*.срр) 
		запоминает путь к файлу, относительно параметра 
		"стартовая папка", и добавляет к нему строку " /"
    reversed1 - 
		для всех файлов запоминает путь к файлу, относительно 
		параметра "стартовая папка", при этом строка пути 
		должна быть преобразована таким образом, чтобы она 
		читалась от имени файла до родительского каталога 
		(например, f\bla\ra\t.dat -> t.dat\ra\bla\f)
    reversed2 - 
		для всех файлов запоминает путь к файлу, относительно параметра
		"стартовая папка", при этом строка пути должна быть перевернута
		(например, f\bla\ra\t.dat -> tad.t\ar\alb\f)

  result:

		путь к файлу с результатами (по умолчанию results.txt)
		
  example: direx.exe c:\some-dir all
